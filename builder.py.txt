"""
graph/builder.py
Builds a LangGraph StateGraph dynamically from workflow.graph_config JSONB.
"""

import logging
from typing import Any

from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.memory import MemorySaver

from graph.state import AgentState
from graph.nodes import (
    find_company_node,
    company_details_node,
    generate_letter_node,
)

logger = logging.getLogger(__name__)

# ── Node registry ────────────────────────────────────────────
# Maps skill.name (LangGraph node ID) → async node function
NODE_REGISTRY: dict[str, Any] = {
    "find-company":    find_company_node,
    "company-details": company_details_node,
    "generate-letter": generate_letter_node,
}


def build_graph(
    graph_config: dict,
    interrupt_before: list[str] | None = None,
    checkpointer=None,
    store=None,
):
    nodes = graph_config.get("nodes", [])

    if not nodes:
        raise ValueError("graph_config has no nodes — agent has no workflow skills configured")

    active_nodes = [n for n in nodes if n.get("enabled", True)]

    if not active_nodes:
        raise ValueError("All workflow steps are disabled — enable at least one step")

    builder = StateGraph(AgentState)

    for node in active_nodes:
        node_id = node["id"]
        if node_id not in NODE_REGISTRY:
            logger.warning(f"Node '{node_id}' not in NODE_REGISTRY — skipping")
            continue
        builder.add_node(node_id, NODE_REGISTRY[node_id])
        logger.debug(f"Added node: {node_id}")

    entry_added = False
    for node in active_nodes:
        node_id = node["id"]
        if node_id not in NODE_REGISTRY:
            continue

        if node.get("is_entry_point") and not entry_added:
            builder.add_edge(START, node_id)
            entry_added = True
            logger.debug(f"START → {node_id}")

        if node.get("is_terminal"):
            builder.add_edge(node_id, END)
            logger.debug(f"{node_id} → END")
            continue

        next_id   = node.get("next")
        condition = node.get("condition")

        if next_id and next_id in NODE_REGISTRY:
            if condition:
                def make_condition(cond: str, nxt: str):
                    def route(state: AgentState) -> str:
                        try:
                            result = eval(cond, {}, dict(state))
                            return nxt if result else END
                        except Exception as e:
                            logger.warning(f"Condition eval failed ({cond}): {e}")
                            return nxt
                    return route
                builder.add_conditional_edges(
                    node_id,
                    make_condition(condition, next_id),
                    {next_id: next_id, END: END}
                )
            else:
                builder.add_edge(node_id, next_id)
                logger.debug(f"{node_id} → {next_id}")
        else:
            builder.add_edge(node_id, END)
            logger.debug(f"{node_id} → END (default)")

    if not entry_added and active_nodes:
        first_id = active_nodes[0]["id"]
        if first_id in NODE_REGISTRY:
            builder.add_edge(START, first_id)
            logger.debug(f"START → {first_id} (fallback)")

    compile_kwargs: dict[str, Any] = {}

    if checkpointer:
        compile_kwargs["checkpointer"] = checkpointer

    if store:
        compile_kwargs["store"] = store

    if interrupt_before:
        valid_interrupts = [
            n for n in interrupt_before
            if n in NODE_REGISTRY and any(node["id"] == n for node in active_nodes)
        ]
        if valid_interrupts:
            compile_kwargs["interrupt_before"] = valid_interrupts
            logger.info(f"Interrupt before: {valid_interrupts}")

    graph = builder.compile(**compile_kwargs)
    logger.info(f"Graph compiled — {len(active_nodes)} active nodes")
    return graph


def build_graph_from_agent(agent: dict, checkpointer=None, store=None):
    graph_config     = agent.get("graph_config") or {}
    interrupt_before = agent.get("interrupt_before") or []

    if not graph_config:
        workflow_skills = agent.get("workflow_skills") or []
        if not workflow_skills:
            raise ValueError(
                f"Agent '{agent.get('name')}' has no workflow skills configured."
            )
        graph_config = _build_config_from_skills(workflow_skills)

    if not checkpointer:
        logger.warning("No checkpointer — using MemorySaver")
        checkpointer = MemorySaver()

    return build_graph(
        graph_config=graph_config,
        interrupt_before=interrupt_before,
        checkpointer=checkpointer,
        store=store,
    )


def _build_config_from_skills(workflow_skills: list[dict]) -> dict:
    nodes = []
    for i, skill in enumerate(workflow_skills):
        next_skill = None
        if i + 1 < len(workflow_skills):
            next_skill = (
                workflow_skills[i + 1].get("skill_name")
                or workflow_skills[i + 1].get("name")
            )
        nodes.append({
            "id":             skill.get("skill_name") or skill.get("name"),
            "is_entry_point": skill.get("is_entry_point", i == 0),
            "is_terminal":    skill.get("is_terminal", i == len(workflow_skills) - 1),
            "next":           skill.get("next_skill_id") or next_skill,
            "condition":      skill.get("condition"),
            "enabled":        skill.get("enabled", True),
        })
    return {"nodes": nodes}
